# pull image
docker pull ghcr.io/mlflow/mlflow:latest

# create folder
mkdir mlflow-data
cd mlflow-data

# run container
docker run -d \
  --name mlflow-server \
  --restart unless-stopped \
  -p 5000:5000 \
  -v ~/mlflow-data:/mlflow \
  ghcr.io/mlflow/mlflow:latest \
  mlflow server \
  --host 0.0.0.0 \
  --port 5000 \
  --allowed-hosts "*" \
  --backend-store-uri sqlite:////mlflow/mlflow.db \
  --default-artifact-root /mlflow/artifacts
